// exmaple prorgram to loop through n numbers and print it

int main()
{
    int n, i =1;
    
    cout <<" enter n" ;
    cin >> n;
    
    while(i<=n)
    {
            cout << i << endl;
            i++;
    }
}

// do while

int main()
{
    int n, i =1;
    
    cout <<" enter n" ;
    cin >> n;
    
    do
    {
            cout << i << endl;
            i++;
    } while(i<=n)
}
